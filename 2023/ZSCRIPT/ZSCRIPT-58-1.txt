class BMNullShellRank3 : Actor
{
	action void A_NullDeathRank3()
    {	
		actor act = null;
		int x = pos.x;
		int y = pos.y;
		int z = pos.z;
		ACS_NamedExecute("increase_subboss_kill_count");
		act = Spawn("BMNullShellRank2", (x + 20, y, z));
		act.ChangeTid(105);
		act = Spawn("BMNullShellRank2", (x - 20, y, z));
		act.ChangeTid(105);
		act = Spawn("BMNullShellRank2", (x, y + 20, z));
		act.ChangeTid(105);
		act = Spawn("BMNullShellRank2", (x, y - 20, z));
		act.ChangeTid(105);
